//过滤对象

```js
  var obj= {
    偃师市: '80%',
    伊川县: '80%',
    洛宁县: '80%',
    宜阳县: '80%',
    汝阳县: '80%',
  }

  var y = {}
  for (const key in obj) {
   y[key] = obj[key].split('%')[0]
  }

  console.log(y)
    最终结果：
   y={
    偃师市: '80',
    伊川县: '80',
    洛宁县: '80',
    宜阳县: '80',
    汝阳县: '80',
   }
```

```js
	var index = 0
			var y = []
			var mapTooltip = {
                "偃师市": "1",
                "伊川县": "9",
                "洛宁县": "34",
              }
			var tooltipColor = ['#72CFDA', '#0648B3', '#ADDE62']
			for (const key in mapTooltip) {
				y[index] = { 
                             name: key,
                             value: mapTooltip[key],
                             color: tooltipColor[index]
                           }
				index++
			}
```
